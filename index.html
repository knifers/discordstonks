async function loadStonks() { try { const response = await fetch('stonks.json'); if (!response.ok) throw new Error("Błąd pobierania danych"); const data = await response.json(); // Dodajemy datę aktualizacji na górze document.querySelector('header p').innerText = `Ostatnia aktualizacja: ${data.updated_at} | Real-time Hype Tracker`; renderChart(data); renderCards(data.apps); } catch (error) { console.error(error); document.getElementById('tickers-grid').innerHTML = '
Błąd: Nie udało się pobrać notowań. Sprawdź plik stonks.json!
'; } } function renderCards(apps) { const container = document.getElementById('tickers-grid'); container.innerHTML = ''; apps.forEach(app => { const isPositive = app.change_24h > 0; const isNegative = app.change_24h < 0; const colorClass = isPositive ? 'text-green-400' : (isNegative ? 'text-red-400' : 'text-slate-500'); const sign = isPositive ? '+' : ''; // Budujemy etykiety źródeł (GitHub / Reddit) let badges = ''; if (app.github_repo) badges += `GITHUB`; if (app.reddit_sub) badges += `r/${app.reddit_sub}`; const html = `
${app.ticker}

${app.name}

${app.current_price.toLocaleString()}
${sign}${app.change_24h}%
${badges}
`; container.innerHTML += html; }); } function renderChart(data) { const ctx = document.getElementById('mainChart').getContext('2d'); const datasets = data.apps.map(app => ({ label: app.name, data: app.history, borderColor: app.color, backgroundColor: 'transparent', borderWidth: 2, tension: 0.3, pointRadius: 0, pointHoverRadius: 6 })); new Chart(ctx, { type: 'line', data: { labels: data.history_dates, datasets: datasets }, options: { responsive: true, maintainAspectRatio: false, interaction: { mode: 'index', intersect: false, }, plugins: { legend: { labels: { color: '#64748b', boxWidth: 10 } }, tooltip: { backgroundColor: '#1e293b', titleColor: '#fff', bodyColor: '#94a3b8', borderColor: '#334155', borderWidth: 1 } }, scales: { y: { grid: { color: '#1e293b' }, ticks: { color: '#64748b' } }, x: { grid: { display: false }, ticks: { color: '#64748b' } } } } }); } loadStonks();
